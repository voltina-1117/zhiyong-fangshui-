```javascript
  function getDeviceFingerprint() {
    try {
      var ua = navigator.userAgent;
      var platform = navigator.platform;
      var language = navigator.language;
      var timezone = new Date().getTimezoneOffset();
      var screenRes = screen.width + 'x' + screen.height;
      var colorDepth = screen.colorDepth;
      
      // 拼接字符串
      var str = ua + platform + language + timezone + screenRes + colorDepth;
      
      // 简单哈希算法
      var hash = 0;
      for (var i = 0; i < str.length; i++) {
        hash = ((hash << 5) - hash) + str.charCodeAt(i);
        hash = hash & hash; // Convert to 32bit
      }
      return Math.abs(hash).toString(36);
    } catch (e) {
      // 降级方案
      return 'fp_' + Math.random().toString(36).substr(2, 9);
    }
  }

  // 检查是否允许提交（防重复）
  function checkFrequencyLimit() {
    var fingerprint = getDeviceFingerprint();
    var lastSubmitKey = 'fs_last_submit_' + fingerprint;
    var lastSubmitTime = parseInt(localStorage.getItem(lastSubmitKey) || '0');
    var now = Date.now();
    var elapsedMs = now - lastSubmitTime;
    var elapsedMin = Math.round(elapsedMs / 1000 / 60);
    
    // 10分钟内不允许重复提交
    if (elapsedMs > 0 && elapsedMs < 10 * 60 * 1000) {
      var waitMin = Math.ceil((10 * 60 * 1000 - elapsedMs) / 1000 / 60);
      return {
        allowed: false,
        reason: '提交过于频繁，请' + waitMin + '分钟后再试'
      };
    }
    
    return { allowed: true };
  }

  // 记录提交时间
  function recordSubmitTime() {
    var fingerprint = getDeviceFingerprint();
    var lastSubmitKey = 'fs_last_submit_' + fingerprint;
    localStorage.setItem(lastSubmitKey, Date.now().toString());
  }

    var el = document.querySelector('[data-err="' + field + '"]');
    if (el) el.textContent = msg || '';

    // 长度检查
    if (trimmed.length < 2 || trimmed.length > 20) {
      return { valid: false, msg: '姓名长度2-20字符' };
    }
    
    // 黑名单关键词
    var blacklist = ['测试', '试试', '123', 'test', 'admin', '删除', '不要', 'xxx'];
    for (var i = 0; i < blacklist.length; i++) {
      if (trimmed.indexOf(blacklist[i]) >= 0) {
        return { valid: false, msg: '姓名包含非法关键词' };
      }
    }
    
    // 字符类型检查
    // 允许：中文 + 英文 + 数字 + 常见符号
    var pattern = /^[\u4e00-\u9fff\u3400-\u4dbf\uF900-\uFAFFA-Za-z0-9\s·\-]{2,20}$/;
    if (!pattern.test(trimmed)) {
      return { valid: false, msg: '姓名格式不正确' };
    }
    
    return { valid: true };
  }

    for (var i = 0; i < blacklist.length; i++) {

        return { valid: false, msg: '描述包含非法关键词' };
      }

    return { valid: true };

      ok = false;
    } else {
      var nameCheck = validateName(data.name);
      if (!nameCheck.valid) {
        setErr('name', nameCheck.msg);
        ok = false;
      }
    }

    // 电话检查

      setErr('phone', '请填写正确的11位手机号码');
      ok = false;
    }

    // 位置检查
    if (!data.location) {
      setErr('location', '请填写所在位置');
      ok = false;
    }

    // 紧急程度检查
    if (!data.urgency) {
      setErr('urgency', '请选择紧急程度');
      ok = false;
    }

    // 描述检查（改进）
    if (!data.description) {
      setErr('description', '请描述现场情况');
      ok = false;
    } else {
      var descCheck = validateDescription(data.description);
      if (!descCheck.valid) {
        setErr('description', descCheck.msg);
        ok = false;
      }
    }

    // 同意条款检查
    if (!document.getElementById('agree').checked) {
      setErr('agree', '请阅读并勾选同意条款');
      ok = false;
    }

    return ok ? data : null;
  }

    return new URLSearchParams(location.search).get('from') || 'direct';
  }

  // ============ 改进：提交逻辑 ============
  form.addEventListener('submit', async function (e) {
    e.preventDefault();
    var data = validate();
    if (!data) return;

    submitBtn.disabled = true;
    submitBtn.textContent = '提交中…';

    try {
      var leadNo;

      if (window.DemoStore.isEnabled()) {
        var lead = window.DemoStore.createLead(Object.assign({}, data, { source: getSource() }));
        leadNo = lead.lead_no;
      } else {
        var res = await fetch(cfg.CREATE_LEAD_URL, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'Authorization': 'Bearer ' + cfg.SUPABASE_ANON_KEY,
          },
          body: JSON.stringify(Object.assign({}, data, { source: getSource() })),
        });
        var json = await res.json();
        if (!res.ok) throw new Error(json.error || '提交失败');
        leadNo = json.lead_no;
      }

      // 记录提交时间（防重复）
      recordSubmitTime();

      location.href = 'success.html?lead_no=' + encodeURIComponent(leadNo);

      toast(err.message || '网络异常，请稍后重试');

    }
  });
})();
```

```javascript
// ============ CSRF防护 ============
var CSRF_TOKEN_KEY = 'fs_csrf_token';

function generateCSRFToken() {
  // 生成随机令牌
  var token = 'csrf_' + 
    Date.now().toString(36) + 
    '_' + 
    Math.random().toString(36).substr(2, 9);
  sessionStorage.setItem(CSRF_TOKEN_KEY, token);
  return token;
}

function getCSRFToken() {
  var token = sessionStorage.getItem(CSRF_TOKEN_KEY);
  if (!token) {
    token = generateCSRFToken();
  }
  return token;
}

// 页面加载时生成令牌
getCSRFToken();

// ============ END CSRF防护 ============
```

**服务端验证** (`supabase/functions/create-lead/index.ts`)：

```typescript
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

const cors = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type, x-csrf-token',
};

const URGENCY_PRIORITY: Record<string, number> = {
  '紧急（需尽快处理）': 1,
  '较急（希望这几天看）': 2,
  '一般（可预约时间）': 3,
};

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: cors });

  try {
    // 验证CSRF令牌（简单版本）
    const csrfToken = req.headers.get('x-csrf-token');
    if (!csrfToken || !csrfToken.startsWith('csrf_')) {
      return json({ error: '请求验证失败' }, 403);
    }

    const body = await req.json();
    const { name, phone, location, urgency, description, source } = body;

    if (!name?.trim() || !/^1[3-9]\d{9}$/.test(phone) || !location?.trim() || !urgency || !description?.trim()) {
      return json({ error: '请填写完整信息' }, 400);
    }

    const supabase = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!,
    );

    const tenMinAgo = new Date(Date.now() - 10 * 60 * 1000).toISOString();
    const { data: recent } = await supabase
      .from('leads')
      .select('id')
      .eq('phone', phone)
      .gte('created_at', tenMinAgo)
      .limit(1);

    if (recent?.length) {
      return json({ error: '提交过于频繁，请10分钟后再试' }, 429);
    }

    const today = new Date().toISOString().slice(0, 10);
    const { data: seqRow } = await supabase
      .from('lead_daily_seq')
      .select('seq')
      .eq('day', today)
      .maybeSingle();

    let seq = (seqRow?.seq ?? 0) + 1;
    await supabase.from('lead_daily_seq').upsert({ day: today, seq });

    const leadNo = `YX${today.replace(/-/g, '')}-${String(seq).padStart(3, '0')}`;
    const priority = URGENCY_PRIORITY[urgency] ?? 3;

    const { data: lead, error } = await supabase
      .from('leads')
      .insert({
        lead_no: leadNo,
        name: name.trim(),
        phone,
        location: location.trim(),
        urgency,
        priority,
        description: description.trim(),
        source: source || 'direct',
        status: '意向待联系',
      })
      .select()
      .single();

    if (error) throw error;

    await notifyServerChan(lead);

    return json({ lead_no: leadNo, id: lead.id }, 200, cors);
  } catch (e) {
    console.error(e);
    return json({ error: e.message || '服务器错误' }, 500, cors);
  }
});
```

```sql
BEFORE UPDATE ON leads
```

## 常见问题排查

### 问题1：提交表单后报"CSRF验证失败"错误
**原因：** 服务端没有更新接收CSRF令牌的逻辑  
**解决：** 确保已更新 `index.ts` 中的服务端代码

### 问题2：设备指纹经常变化
**原因：** UA或其他参数在变化  
**解决：** 这是正常的，浏览器关闭后会重置。可以改用 localStorage 中的持久化指纹

### 问题3：表单在某些浏览器上输入框被放大
**原因：** iOS Safari 的默认行为（<16px字体会自动放大）  
**解决：** 确保 `font-size: 16px !important` 已应用于所有输入框

### 问题4：Server酱没有收到推送
**原因：** SendKey 配置错误或过期  
**解决：** 
1. 重新访问 https://sct.ftqq.com
2. 确认SendKey是否正确
3. 尝试手动调用Server酱API测试
