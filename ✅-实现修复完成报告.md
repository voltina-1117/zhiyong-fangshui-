```typescript
    'authorization, x-client-info, apikey, content-type, x-csrf-token',  // ← 新增
```

```javascript
// js/admin.js
var CSRF_TOKEN_KEY = 'admin_csrf_token';
function getCSRFToken() {
  var token = sessionStorage.getItem(CSRF_TOKEN_KEY);
  if (!token) {
    token = 'csrf_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
    sessionStorage.setItem(CSRF_TOKEN_KEY, token);
  }
  return token;
}

// 所有API调用时添加
async function api(action, payload) {
  var res = await fetch(cfg.ADMIN_API_URL, {
    headers: {
      'X-CSRF-Token': getCSRFToken(),  // ← 新增
    },
    body: JSON.stringify(Object.assign({ action, token }, payload || {})),
  });

}
```

## 现在的表单流程

```
1. 用户打开表单
   ↓
2. 前端生成设备指纹 + CSRF令牌
   ↓
3. 用户填写信息（显示进度条）
   ↓
4. 用户快速重复点击提交
   ↓
5. 前端检查频率限制 → 阻止
   ↓
6. 用户真正提交
   ↓
7. 前端检查频率限制 → 允许
   ↓
8. 发送请求（带CSRF token）
   ↓
9. 后端验证CSRF token → 通过
   ↓
10. 后端检查手机号10分钟内是否重复 → 阻止脚本批量提交
   ↓
11. 创建意向单 + 发送Server酱通知
```
